159 void PressureController::compute(uint16_t p_centiSec) {
160  updateBlower(p_centiSec);
161 
162  // Update the cycle phase
163  updatePhase(p_centiSec);
164 
165  if (!m_vigilance) {
166  // Act accordingly
167  switch (m_subPhase) {
168  case CycleSubPhases::INSPIRATION: {
169  inhale();
170  break;
171  }
172  case CycleSubPhases::HOLD_INSPIRATION: {
173  plateau();

200 void PressureController::computePlateau(uint16_t p_centiSec) {
201  uint16_t minValue = m_lastPressureValues[0u];
202  uint16_t maxValue = m_lastPressureValues[0u];
203  uint16_t totalValues = m_lastPressureValues[0u];
204 
205  for (uint8_t index = 1u; index < MAX_PRESSURE_SAMPLES; index++) {
206  minValue = min(minValue, m_lastPressureValues[index]);
207  maxValue = max(maxValue, m_lastPressureValues[index]);
208  totalValues += m_lastPressureValues[index];
209  }
210 
211  uint16_t diff = (maxValue - minValue);
212 
213  // Start computing plateau pressure when:
214  // - the last pressure values were close enough
215  // - the hold inspiration phase is about to end
216  // - plateau pressure computation was not already started
217  if (!m_plateauComputed && (diff < 10u)
218  && (p_centiSec >= ((m_centiSecPerInhalation * 95u) / 100u))) {
219  m_startPlateauComputation = true;
220  }
221 
222  // Stop computing plateau pressure when pressure drops
223  if (m_startPlateauComputation && (diff > 10u)) {
224  m_startPlateauComputation = false;
225  m_plateauComputed = true;
226  }
227 
228  if (m_startPlateauComputation) {
229  m_plateauPressure = totalValues / MAX_PRESSURE_SAMPLES;
230  }
231 }

485 int32_t PressureController::pidBlower(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
486  // Compute error
487  int32_t error = targetPressure - currentPressure;
488 
489  // Compute integral
490  blowerIntegral = blowerIntegral + ((PID_BLOWER_KI * error * dt) / 1000000);
491  blowerIntegral = max(PID_BLOWER_INTEGRAL_MIN, min(PID_BLOWER_INTEGRAL_MAX, blowerIntegral));
492 
493  // Compute derivative
494  int32_t derivative = ((blowerLastError == INVALID_ERROR_MARKER) || (dt == 0))
495  ? 0
496  : ((1000000 * (error - blowerLastError)) / dt);
497  blowerLastError = error;
498 
499  int32_t blowerCommand = (PID_BLOWER_KP * error) + blowerIntegral
500  + ((PID_BLOWER_KD * derivative) / 1000); // Command computation
501 
502  int32_t minAperture = m_blower_valve.minAperture();
503  int32_t maxAperture = m_blower_valve.maxAperture();
504 
505  uint32_t blowerAperture =
506  max(minAperture,
507  min(maxAperture, maxAperture + (minAperture - maxAperture) * blowerCommand / 1000));
508 
509  return blowerAperture;
510 }

513 PressureController::pidPatient(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
514  // Compute error
515  int32_t error = targetPressure + PID_PATIENT_SAFETY_PEEP_OFFSET - currentPressure;
516 
517  // Compute integral
518  patientIntegral = patientIntegral + ((PID_PATIENT_KI * error * dt) / 1000000);
519  patientIntegral = max(PID_PATIENT_INTEGRAL_MIN, min(PID_PATIENT_INTEGRAL_MAX, patientIntegral));
520 
521  // Compute derivative
522  int32_t derivative = ((patientLastError == INVALID_ERROR_MARKER) || (dt == 0))
523  ? 0
524  : ((1000000 * (error - patientLastError)) / dt);
525  patientLastError = error;
526 
527  int32_t patientCommand = (PID_PATIENT_KP * error) + patientIntegral
528  + ((PID_PATIENT_KD * derivative) / 1000); // Command computation
529 
530  int32_t minAperture = m_blower_valve.minAperture();
531  int32_t maxAperture = m_blower_valve.maxAperture();
532 
533  uint32_t patientAperture =
534  max(minAperture,
535  min(maxAperture, maxAperture + (maxAperture - minAperture) * patientCommand / 1000));
